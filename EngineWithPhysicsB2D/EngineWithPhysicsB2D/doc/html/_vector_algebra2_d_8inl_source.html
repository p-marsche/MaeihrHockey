62 template <typename T>
63 void setPolarAngle(sf::Vector2<T>& vector, T newAngle)
64 {
65 // No assert here, because turning a zero vector is well-defined (yields always zero vector)
66
67 T vecLength = length(vector);
68
69 vector.x = vecLength * TrigonometricTraits<T>::cos(newAngle);
70 vector.y = vecLength * TrigonometricTraits<T>::sin(newAngle);
71 }

73 template <typename T>
74 void rotate(sf::Vector2<T>& vector, T angle)
75 {
76 // No assert here, because turning a zero vector is well-defined (yields always zero vector)
77
78 T cos = TrigonometricTraits<T>::cos(angle);
79 T sin = TrigonometricTraits<T>::sin(angle);
80
81 // Don't manipulate x and y separately, otherwise they're overwritten too early
82 vector = sf::Vector2<T>(
83 cos * vector.x - sin * vector.y,
84 sin * vector.x + cos * vector.y);
85 }

108 {
109 assert(lhs != sf::Vector2<T>() && rhs != sf::Vector2<T>());
